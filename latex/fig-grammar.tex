\setlength{\grammarindent}{5em}
\begin{grammar}
	<Expr> ::= z | \CC\ | sin(<Expr>) | cos(<Expr>) | exp(<Expr>)
	\alt log(<Expr>) | <Sum> | <Term>

	<Sum> ::= <Expr> $\times$ \CC\ | <Expr> $\times$ \CC\ + <Sum>

	<Term> ::= <Expr> \textsuperscript{\CC} | <Expr> \textsuperscript{\CC}  $\times$ <Term>
\end{grammar}
